215  template <typename UMixer>
216  typename std::enable_if<(UMixer::block_bits > 0), size_type>::type
217  count_impl(UMixer& mixer) const
218  {
219  static_assert(UMixer::block_bits / (sizeof(block_type) * 8),
220  "storage_type::block_type not compatible with mixer_type bit alignment"
221  );
222  constexpr auto storage_blocks_in_mixer_block = UMixer::block_bits / (sizeof(block_type) * 8);
223  block_type blocks[storage_blocks_in_mixer_block];
224  bool block_loaded[storage_blocks_in_mixer_block];
225  for (std::size_t idx = 0; idx < storage_blocks_in_mixer_block; ++idx)
226  block_loaded[idx] = false;
227 
228  std::size_t base_bit_offset = mixer.base_offset();
229  std::size_t base_block_offset = base_bit_offset / (sizeof(block_type) * 8);
230  for (std::size_t count = 0; count < _params.num_hashes; ++count)
231  {
232  auto inner_bit_idx = mixer() - base_bit_offset;
233  auto inner_block_idx = inner_bit_idx / (8 * sizeof(block_type));
234  auto inner_block_bit = inner_bit_idx % (8 * sizeof(block_type));
235  if (!block_loaded[inner_block_idx])
236  {
237  blocks[inner_block_idx] = _data[base_block_offset + inner_block_idx];
238  block_loaded[inner_block_idx] = true;
239  }
240 
241  if (!(blocks[inner_block_idx] & (block_type(1) << inner_block_bit)))
242  return 0;
243  }
244  return 1;
245  }
